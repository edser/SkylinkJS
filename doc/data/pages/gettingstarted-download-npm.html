Temasys SDK is also available on npm and bower. Append the @0.7.x tag (e.g. skylinkjs@0.7.x) to install the latest 0.7.x version, or the version number (e.g. skylinkjs@0.7.1) to install custom SDK version.

For npm usage: 

npm install skylinkjs

For bower usage: 

bower install skylinkjs

Code Usage Example:

var AdapterJS = require("adapterjs");
var io = require("socket.io-client");
var Temasys = require("skylinkjs").Temasys;

// Deprecated Usage
var Skylink = require("skylinkjs").Skylink;
var SkylinkLogs = require("skylinkjs").SkylinkLogs;

Replace the require("adapterjs") to reference to any customized Temasys AdapterJS for integrating customized Temasys WebRTC plugin and Screensharing extensions.
